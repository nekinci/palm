# palm
[]: # Version: 0.0.1
[]: # Authors: [Niyazi Ekinci](https://github.com/nekinci)

Palm is an programming language where you can write your code in a way that you can understand it easily.

There will be a specification for palm language as soon as possible.


```palm
    // Type aliases
    type abc = struct {
        int value
        
    }
    

    // Değişken tanımlama
    a := 5 // a is integer and it's value is 5
    b := 5.5 // b is float and it's value is 5.5
    c := "Hello World" // c is string and it's value is "Hello World"
    d := true // d is boolean and it's value is true
    e := false // e is boolean and it's value is false
    f := [1, 2, 3] // f is int array and it's value is [1, 2, 3]
    g := [1.5, 2.5, 3.5] // g is float array and it's value is [1.5, 2.5, 3.5]
    h := ["Hello", "World"] // h is string array and it's value is ["Hello", "World"]
    i := [true, false] // i is boolean array and it's value is [true, false]
    
    // ya da şu şekilde de tanımlanabilir
    int a = 5
    float b = 5.5
    string c = "Hello World"
    bool d = true
    bool e = false
    int[] f = [1, 2, 3]
    float[] g = [1.5, 2.5, 3.5]
    string[] h = ["Hello", "World"]
    bool[] i = [true, false]
    interface j = 5 // j is interface and it's value is 5
    interface[] k = [1, 2, 3, true, false, "five"] // k is interface array and it's value is [1, 2, 3]
    
    // Değişkenlerin değerlerini değiştirmek için
    a = 10 // a is integer and it's value is 10
    b = 10.5 // b is float and it's value is 10.5
    c = "Hello Palm" // c is string and it's value is "Hello Palm"
    d = false // d is boolean and it's value is false
    e = true // e is boolean and it's value is true
    f = [4, 5, 6] // f is int array and it's value is [4, 5, 6]
    g = [4.5, 5.5, 6.5] // g is float array and it's value is [4.5, 5.5, 6.5]
    h = ["Hello", "Palm"] // h is string array and it's value is ["Hello", "Palm"]
    j = "10" // j is interface and it's value is "10"
    k = [4, 5, 6, false, true, "six"] // k is interface array and it's value is [4, 5, 6, false, true, "six"]
```
